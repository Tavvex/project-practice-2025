Алгоритмическая задача "Выставление тегов"

def main():
    n = int(input())
    if n == 1:
        print(1)
    elif n == 2:
        print(2)
    else:
        tags = [0] * 36
        tags[1] = 1
        tags[2] = 1
        for i in range(3,n+1):
            tags[i] = tags[i-1] + tags[i-2]
        print(sum(tags))
if __name__ == '__main__':
    main()
